2. Anfrage:
(
  SELECT 'Buch' AS Typ, p.PNr AS ProduktNr, p.Rating, COUNT(r.RezensionsID) AS AnzahlBewertungen
  FROM Produkt p
  JOIN Buch b ON p.PNr = b.PNr
  LEFT JOIN Rezension r ON p.PNr = r.PNr
  WHERE p.Rating IS NOT NULL
  GROUP BY p.PNr, p.Rating
  ORDER BY p.Rating DESC, COUNT(r.RezensionsID) DESC, p.PNR
  LIMIT 5
)
UNION ALL
(
  SELECT 'Musik-CD' AS Typ, p.PNr AS ProduktNr, p.Rating, COUNT(r.RezensionsID)
  FROM Produkt p
  JOIN Musik_CD m ON p.PNr = m.PNr
  LEFT JOIN Rezension r ON p.PNr = r.PNr
  WHERE p.Rating IS NOT NULL
  GROUP BY p.PNr, p.Rating
  ORDER BY p.Rating DESC, COUNT(r.RezensionsID) DESC, p.PNR
  LIMIT 5
)
UNION ALL
(
  SELECT 'DVD' AS Typ, p.PNr AS ProduktNr, p.Rating, COUNT(r.RezensionsID)
  FROM Produkt p
  JOIN DVD d ON p.PNr = d.PNr
  LEFT JOIN Rezension r ON p.PNr = r.PNr
  WHERE p.Rating IS NOT NULL
  GROUP BY p.PNr, p.Rating
  ORDER BY p.Rating DESC, COUNT(r.RezensionsID) DESC, p.PNR
  LIMIT 5
);

4. Anfrage:
select pnr, max(preis) as maximalpreis, min(preis) as minimalpreis
from angebot a
where a.preis > 0.00
group by pnr
having max(preis) > (min(preis) * 2)


5. Anfrage:
select pnr, max(bewertung) as beste_bewertung, min(bewertung) as schlechteste_bewertung
from rezension
group by pnr
having max(bewertung) = 5 and min(bewertung) = 1

7. Anfrage:
select nutzername, count(rezensionsid) as anzahl_rezensionen
from rezension
where nutzername <> 'guest'
group by nutzername
having count(rezensionsid) > 9
order by anzahl_rezensionen desc

8. Anfrage:
SELECT DISTINCT a.Name
FROM Autor a
WHERE a.Name IN (
    SELECT Name FROM Creator
    UNION ALL
    SELECT Name FROM Director
    UNION ALL
    SELECT Name FROM Actor
    UNION ALL
    SELECT Kuenstlername AS Name FROM Kuenstler
)
ORDER BY a.Name;

10. Anfrage (Rekursion):

WITH RECURSIVE hauptkategorien AS (
  -- TOP DOWN: an Wurzel (hauptkat.) beginnen, in jeder Iteration direkte unterkategorien finden,
  -- jeweils die Hauptkategorie ihrer direkten oberkategorie als Hauptkategorie eintragen
  SELECT
    KategorieID,
    KategorieID AS HauptkategorieID
  FROM Kategorie
  WHERE OberkategorieID IS NULL --das liefert ausschliesslich HAUPTKATEGORIEN

  UNION ALL

  -- Rekursion: direkte Unterkategorien abfragen
  SELECT
    k.KategorieID,
    hk.HauptkategorieID
  FROM Kategorie k
  JOIN hauptkategorien hk
  ON k.OberkategorieID = hk.KategorieID
),

produkt_hk AS (
	SELECT pkat.pnr, (hkat.hauptkategorieID) AS hk
	FROM hauptkategorien hkat
	JOIN produkt_kategorie pkat
	ON hkat.kategorieID = pkat.kategorieID
)

SELECT DISTINCT p1.PNr
FROM produkt_hk p1
JOIN Produkt_Aehnlichkeit pa ON p1.PNr = pa.PNr1
JOIN produkt_hk p2 ON pa.PNr2 = p2.PNr
WHERE NOT p1.hk = p2.hk;

12. Anfrage:
with produkte_in_allen_filialen as (
select distinct pnr
from angebot
where preis > 0
group by pnr
having count(distinct filialeid) = (select count(*) from filiale)
),
günstigstes_angebot as (
select distinct pnr, min(preis) as min_preis
from angebot
where pnr in (select pnr from produkte_in_allen_filialen) and preis > 0.00
group by pnr
),
günstigstes_angebot_in_leipzig as (
select distinct a.pnr
from angebot a join filiale f on a.filialeid = f.filialeid join günstigstes_angebot ga on a.pnr = ga.pnr and a.preis = ga.min_preis
where f.name = 'Leipzig'
)
select count(paf.pnr) as anzahl_produkte_alle_filialen,
	   count(gal.pnr) as anzahl_günstigste_in_leipzig,
	  round(count(gal.pnr) * 100.00 / count(paf.pnr)) as anteil_günstigstes_in_leipzig
from produkte_in_allen_filialen paf left join günstigstes_angebot_in_leipzig gal on paf.pnr = gal.pnr
