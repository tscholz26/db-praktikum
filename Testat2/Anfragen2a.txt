1. Anfrage: Wieviele Produkte jeden Typs (Buch, Musik-CD, DVD) sind in der Datenbank erfasst? Hinweis: Geben Sie das Ergebnis in einer 3-spaltigen Relation aus.
select
    (select count(*) as Anzahl_Bücher from buch),
    (select count(*) as Anzahl_DVDs from dvd),
    (select count(*) as Anzahl_MusikCD from musik_cd)


2. Anfrage: Nennen Sie die 5 besten Produkte jedes Typs (Buch, Musik-CD, DVD) sortiert nach dem durchschnittlichem Rating. Hinweis: Geben Sie das Ergebnis in einer einzigen Relation mit den Attributen Typ, ProduktNr, Rating aus. Wie werden gleiche durchschnittliche Ratings behandelt?
(
  SELECT 'Buch' AS Typ, p.PNr AS ProduktNr, p.Rating, COUNT(r.RezensionsID) AS AnzahlBewertungen
  FROM Produkt p
  JOIN Buch b ON p.PNr = b.PNr
  LEFT JOIN Rezension r ON p.PNr = r.PNr
  WHERE p.Rating IS NOT NULL
  GROUP BY p.PNr, p.Rating
  ORDER BY p.Rating DESC, COUNT(r.RezensionsID) DESC, p.PNR
  LIMIT 5
)
UNION ALL
(
  SELECT 'Musik-CD' AS Typ, p.PNr AS ProduktNr, p.Rating, COUNT(r.RezensionsID)
  FROM Produkt p
  JOIN Musik_CD m ON p.PNr = m.PNr
  LEFT JOIN Rezension r ON p.PNr = r.PNr
  WHERE p.Rating IS NOT NULL
  GROUP BY p.PNr, p.Rating
  ORDER BY p.Rating DESC, COUNT(r.RezensionsID) DESC, p.PNR
  LIMIT 5
)
UNION ALL
(
  SELECT 'DVD' AS Typ, p.PNr AS ProduktNr, p.Rating, COUNT(r.RezensionsID)
  FROM Produkt p
  JOIN DVD d ON p.PNr = d.PNr
  LEFT JOIN Rezension r ON p.PNr = r.PNr
  WHERE p.Rating IS NOT NULL
  GROUP BY p.PNr, p.Rating
  ORDER BY p.Rating DESC, COUNT(r.RezensionsID) DESC, p.PNR
  LIMIT 5
);

3. Anfrage: Für welche Produkte gibt es im Moment kein Angebot?

SELECT *
FROM Produkt p
WHERE NOT EXISTS (
    SELECT a.PNr
    FROM Angebot a
    WHERE (a.preis > 0.00) AND (a.PNr = p.PNr)
)

4. Anfrage: Für welche Produkte ist das teuerste Angebot mehr als doppelt so teuer wie das preiswerteste?
select pnr, max(preis) as maximalpreis, min(preis) as minimalpreis
from angebot a
where a.preis > 0.00
group by pnr
having max(preis) > (min(preis) * 2)


5. Anfrage: Welche Produkte haben sowohl mindestens eine sehr schlechte (Punktzahl: 1) als auch mindestens eine sehr gute (Punktzahl: 5) Bewertung?
select pnr, max(bewertung) as beste_bewertung, min(bewertung) as schlechteste_bewertung
from rezension
group by pnr
having max(bewertung) = 5 and min(bewertung) = 1

6. Anfrage: Für wieviele Produkte gibt es gar keine Rezension?
SELECT COUNT(*) AS Anz
FROM (
    SELECT PNr FROM Produkt
    EXCEPT
    SELECT PNr FROM Rezension
) AS keine_rezension;

7. Anfrage: Nennen Sie alle Rezensenten, die mindestens 10 Rezensionen geschrieben haben.
select nutzername, count(rezensionsid) as anzahl_rezensionen
from rezension
where nutzername <> 'guest'
group by nutzername
having count(rezensionsid) > 9
order by anzahl_rezensionen desc

8. Anfrage: Geben Sie eine duplikatfreie und alphabetisch sortierte Liste der Namen aller Buchautoren an, die auch an DVDs oder Musik-CDs beteiligt sind.
SELECT DISTINCT a.Name
FROM Autor a
WHERE a.Name IN (
    SELECT Name FROM Creator
    UNION ALL
    SELECT Name FROM Director
    UNION ALL
    SELECT Name FROM Actor
    UNION ALL
    SELECT Kuenstlername AS Name FROM Kuenstler
)
ORDER BY a.Name;

10. Anfrage: Für welche Produkte gibt es ähnliche Produkte in einer anderen Hauptkategorie? Hinweis: Eine Hauptkategorie ist eine Produktkategorie ohne Oberkategorie. Erstellen Sie eine rekursive Anfrage, die zu jedem Produkt dessen Hauptkategorie bestimmt.
WITH RECURSIVE hauptkategorien AS (
  -- TOP DOWN: an Wurzel (hauptkat.) beginnen, in jeder Iteration direkte unterkategorien finden,
  -- jeweils die Hauptkategorie ihrer direkten oberkategorie als Hauptkategorie eintragen
  SELECT
    KategorieID,
    KategorieID AS HauptkategorieID
  FROM Kategorie
  WHERE OberkategorieID IS NULL --das liefert ausschliesslich HAUPTKATEGORIEN

  UNION ALL

  -- Rekursion: direkte Unterkategorien abfragen
  SELECT
    k.KategorieID,
    hk.HauptkategorieID
  FROM Kategorie k
  JOIN hauptkategorien hk
  ON k.OberkategorieID = hk.KategorieID
),

produkt_hk AS (
	SELECT pkat.pnr, (hkat.hauptkategorieID) AS hk
	FROM hauptkategorien hkat
	JOIN produkt_kategorie pkat
	ON hkat.kategorieID = pkat.kategorieID
)

SELECT DISTINCT p1.PNr
FROM produkt_hk p1
JOIN Produkt_Aehnlichkeit pa ON p1.PNr = pa.PNr1
JOIN produkt_hk p2 ON pa.PNr2 = p2.PNr
WHERE NOT p1.hk = p2.hk;


11. Anfrage: Welche Produkte werden in allen Filialen angeboten? Hinweis: Ihre Query muss so formuliert werden, dass sie für eine beliebige Anzahl von Filialen funktioniert. Hinweis: Beachten Sie, dass ein Produkt mehrfach von einer Filiale angeboten werden kann (z.B. neu und gebraucht).
SELECT PNr
    FROM Angebot
    WHERE preis > 0
    GROUP BY PNr
    HAVING COUNT(DISTINCT FilialeID) = (SELECT COUNT(*) FROM Filiale)

12. Anfrage: In wieviel Prozent der Fälle der Frage 11 gibt es in Leipzig das preiswerteste Angebot?
WITH
produkte_alle_filialen AS (
    SELECT PNr
    FROM Angebot
    WHERE preis > 0
    GROUP BY PNr
    HAVING COUNT(DISTINCT FilialeID) = (SELECT COUNT(*) FROM Filiale)
),

minpreis_global AS (
    SELECT a.PNr, MIN(a.Preis) AS preis
    FROM Angebot a
    WHERE a.Preis > 0
    GROUP BY a.PNr
),

minpreis_leipzig AS (
    SELECT a.PNr, MIN(a.Preis) AS preis
    FROM Angebot a NATURAL JOIN Filiale f
    WHERE f.Name LIKE '%Leipzig%' AND a.Preis > 0
    GROUP BY a.PNr
),

leipzig_am_guenstigsten AS (
    SELECT min_l.PNr, min_l.Preis
    FROM produkte_alle_filialen p
    JOIN minpreis_global min_g ON p.pnr = min_g.pnr
    JOIN minpreis_leipzig min_l ON min_g.pnr = min_l.pnr AND min_g.preis = min_l.preis
)

SELECT
    (SELECT COUNT(*) FROM produkte_alle_filialen) AS anzahl_produkte_alle_filialen,
    (SELECT COUNT(*) FROM leipzig_am_guenstigsten) AS anzahl_leipzig_am_guenstigsten,
    ROUND(
        (SELECT COUNT(*) FROM leipzig_am_guenstigsten) * 100.0 /
        (SELECT COUNT(*) FROM produkte_alle_filialen),
        2
    ) AS anteil_leipzig_am_guenstigsten;
