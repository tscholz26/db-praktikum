2. Anfrage:
create view top_5_produkte_je_typ(typ, pnr, rating)as
(
	select 'buch' as typ, p.pnr, p.rating
	from buch b join produkt p on b.pnr = p.pnr
	where p.rating is not null
	order by p.rating desc
	limit 5
) union
(
	select 'dvd' as typ, p.pnr, p.rating
	from dvd join produkt p on dvd.pnr = p.pnr
	where p.rating is not null
	order by p.rating desc
	limit 5
) union
(	
	select 'Musik_CD' as typ, p.pnr, p.rating
	from musik_cd m join produkt p on m.pnr = p.pnr
	where p.rating is not null
	order by p.rating desc
	limit 5
);

Select typ, pnr, rating
from top_5_produkte_je_typ

Anmerkung: Gleiche durchschnittliche Ratings werden nicht berücksichtigt, d.h. es könnte noch mehrere Produkte mit gleich gutem Rating geben, die aber nicht zurückgegeben werden. Sollten alle Produkte der 5 bestbewertesten Produkte jedes Typs zurückgegeben werden, würde sich eine Anfrage über dense_rank() anbieten

12. Anfrage:
with produkte_in_allen_filialen as (
select pnr
from angebot
group by pnr
having count(distinct filialeid) = (select count(*) from filiale)
),
günstigstes_angebot as (
select pnr, min(preis) as min_preis
from angebot
where pnr in (select pnr from produkte_in_allen_filialen)
group by pnr
),
günstigstes_angebot_in_leipzig as (
select distinct a.pnr
from angebot a join filiale f on a.filialeid = f.filialeid join günstigstes_angebot ga on a.pnr = ga.pnr and a.preis = ga.min_preis
where f.name = 'Leipzig'
)
select count(paf.pnr) as anzahl_produkte_alle_filialen, 
	   count(gal.pnr) as anzahl_günstigste_in_leipzig, 
	  round(count(gal.pnr) * 100.0 / count(paf.pnr)) as anteil_günstigstes_in_leipzig
from produkte_in_allen_filialen paf left join günstigstes_angebot_in_leipzig gal on paf.pnr = gal.pnr
